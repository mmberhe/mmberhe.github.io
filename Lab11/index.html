Q1: Write the necessary Node script to make this code work for all arrays: [1,2,3,4,5,6,7,8].even() prints [2,4,6,8] and [1,2,3,4,5,6,7,8].odd() prints [1,3,5,7]. Test your code in Node.js CLI

        
            Array.prototype.even() = function(){
                return this.filter( x=> x%2==0);
            }


            Array.prototype.odd() = function(){
                return this.filter( x=> x%2!=0);
            }

            console.log([1,2,3,4,5,6,7,8].even()); // [2,4,6,8]
            console.log([1,2,3,4,5,6,7,8].odd()); // [1,3,5,7]


            // we can run it in Terminal using this command "node q1.js"

Q1: Create a webserver that's going to send a response of a pdf file to any client that sends a request to your specified server:port. Set a content-type as "application/pdf".

        
            const http = require("http");
            const fs = require("fs");

            const server = http.createServer((req,res) => {

                res.writeHead(200,{
                    'Content-Type': 'application/pdf',
                });


                fs.readFile('./mypdf.pdf',(err,data)=>{
                    if(err) console.log('File couldn't found);
                    res.end(data);
                })
            });

            const PORT = 3000;
            server.listen(PORT,() => {
                console.log(`Server listening on port${PORT}`);
            })
